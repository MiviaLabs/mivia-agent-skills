func RetryWithBackoff(ctx context.Context, attempts int, fn func() error) error {
	var lastErr error
	for i := 0; i < attempts; i++ {
		if err := ctx.Err(); err != nil {
			return err
		}
		if err := fn(); err != nil {
			lastErr = err
			// Create a timer that can be stopped
			timer := time.NewTimer(time.Duration(i+1) * 100 * time.Millisecond)
			select {
			case <-timer.C:
				// Timer fired, stop it to release resources
				if !timer.Stop() {
					<-timer.C // drain if stopped after firing
				}
			case <-ctx.Done():
				// Context cancelled, stop the timer
				if !timer.Stop() {
					<-timer.C // drain if stopped after firing
				}
				return ctx.Err()
			}
			continue
		}
		return nil
	}
	return lastErr
}
